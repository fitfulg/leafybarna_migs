THIS QUERY FILTERS LOCAL INTEREST TREES BY DISTRICT, FOR EXAMPLE, DISTRICT 7.

Filter by District: 7

SELECT * FROM local_interest_trees WHERE district_id = 7 LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY NEIGHBORHOOD, FOR EXAMPLE, 'HORTA'.

Filter by Neighborhood: Horta

SELECT * FROM local_interest_trees WHERE neighborhood_name ILIKE '%Horta%' LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY NAME, FOR EXAMPLE, 'PI PINYER'.

Filter by Name: Pi Pinyer

SELECT * FROM local_interest_trees WHERE name ILIKE '%Pi Pinyer%' LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY ADDRESS, FOR EXAMPLE, 'PASSEIG DE SANT JOAN'.

Filter by Address: Passeig de Sant Joan


SELECT * FROM local_interest_trees WHERE address_name ILIKE '%Sant Joan%' LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY CREATION DATE, FOR EXAMPLE, '2021-11-24'.

Filter by Creation Date: 2021-11-24


SELECT * FROM local_interest_trees WHERE created = '2021-11-24' LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY BODY CONTENT, FOR EXAMPLE, 'PARC'.

Filter by Body Content: Parc


SELECT * FROM local_interest_trees WHERE body ILIKE '%parc%' LIMIT 10;

THIS QUERY FILTERS LOCAL INTEREST TREES BY BOTH DISTRICT AND NAME, FOR EXAMPLE, DISTRICT 7 AND NAME 'PI PINYER'.

Filter by District 7 and Name: Pi Pinyer

SELECT * FROM local_interest_trees WHERE district_id = 7 AND name ILIKE '%Pi Pinyer%' LIMIT 10;
