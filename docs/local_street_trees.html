\dt

THIS COMMAND SHOWS THE STRUCTURE OF THE LOCAL_STREET_TREES TABLE, INCLUDING THE COLUMNS AND THEIR TYPES.

\d local_street_trees

THIS QUERY FILTERS LOCAL STREET TREES LOCATED IN DISTRICT 8.

Filter by District 8

SELECT * FROM local_street_trees WHERE ROUND(codi_districte::numeric) = 8 LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY SPECIES, FOR EXAMPLE, 'FRAXINUS'.

Filter by Species: Fraxinus

SELECT * FROM local_street_trees WHERE nom_cientific LIKE '%Fraxinus%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY ADDRESS, FOR EXAMPLE, 'PG FABRA I PUIG'.

Filter by Address: Pg Fabra i Puig

SELECT * FROM local_street_trees WHERE adreca LIKE '%Pg Fabra i Puig%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY NEIGHBORHOOD, FOR EXAMPLE, 'NOU BARRIS'.

Filter by Neighborhood: Nou Barris

SELECT * FROM local_street_trees WHERE nom_districte = 'NOU BARRIS' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY THE IRRIGATION TYPE, FOR EXAMPLE, 'GOTEIG'.

Filter by Irrigation: GOTEIG

SELECT * FROM local_street_trees WHERE tipus_reg LIKE '%GOTEIG%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY PLANTING DATE, FOR EXAMPLE, '2020-05-10'.

Filter by Planting Date: 2020-05-10

SELECT * FROM local_street_trees WHERE data_plantacio = '2020-05-10' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY CATALOGATION, FOR EXAMPLE, 'INTERÈS LOCAL'.

Filter by Catalogation: INTERÈS LOCAL

SELECT * FROM local_street_trees WHERE catalogacio LIKE '%INTERÈS LOCAL%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY BOTH DISTRICT AND SPECIES, FOR EXAMPLE, DISTRICT 8 AND SPECIES 'FRAXINUS'.

Filter by District 8 and Species: Fraxinus

SELECT * FROM local_street_trees WHERE ROUND(codi_districte::numeric) = 8 AND nom_cientific LIKE '%Fraxinus%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY NEIGHBORHOOD AND CATALOGATION, FOR EXAMPLE, 'NOU BARRIS' AND 'INTERÈS LOCAL'.

Filter by Neighborhood: Nou Barris and Catalogation: INTERÈS LOCAL

SELECT * FROM local_street_trees WHERE nom_districte = 'NOU BARRIS' AND catalogacio LIKE '%INTERÈS LOCAL%' LIMIT 10;

THIS QUERY FILTERS LOCAL STREET TREES BY A SPECIFIC NEIGHBORHOOD, FOR EXAMPLE, 'VILAPICINA I LA TORRE LLOBETA'.

Filter by Neighborhood: VILAPICINA I LA TORRE LLOBETA

SELECT * FROM local_street_trees WHERE nom_barri = 'VILAPICINA I LA TORRE LLOBETA' LIMIT 10;
