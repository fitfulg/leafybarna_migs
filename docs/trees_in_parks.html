THIS QUERY FILTERS TREES IN PARKS BY CODE, FOR EXAMPLE, '0000024AR'.

Filter by Code: 0000024AR

SELECT * FROM trees_in_parks WHERE codi = '0000024AR' LIMIT 10;

THIS QUERY FILTERS TREES IN PARKS BY SCIENTIFIC NAME, FOR EXAMPLE, 'POPULUS NIGRA'.

Filter by Scientific Name: Populus nigra

SELECT * FROM trees_in_parks WHERE nom_cientific LIKE '%Populus nigra%' LIMIT 10;

THIS QUERY FILTERS TREES IN PARKS BY DISTRICT NAME, FOR EXAMPLE, 'NOU BARRIS'.

Filter by District: Nou Barris

SELECT * FROM trees_in_parks WHERE nom_districte LIKE '%NOU BARRIS%' LIMIT 10;

THIS QUERY FILTERS TREES IN PARKS BY ELEMENT TYPE, FOR EXAMPLE, 'ARBRE PARC'.

Filter by Element Type: ARBRE PARC

THIS QUERY FILTERS TREES IN PARKS BY NEIGHBORHOOD, FOR EXAMPLE, 'LA GUINEUETA'.

Filter by Neighborhood: La Guineueta

SELECT * FROM trees_in_parks WHERE nom_barri LIKE '%LA GUINEUETA%' LIMIT 10;

THIS QUERY FILTERS TREES IN PARKS BY COORDINATES.

Filter by Coordinates
