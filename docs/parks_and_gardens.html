THIS QUERY FILTERS PARKS AND GARDENS LOCATED IN DISTRICT 7.

Filter by District 7

SELECT * FROM parks_and_gardens WHERE district_id = 7 LIMIT 10;

THIS QUERY FILTERS PARKS AND GARDENS BY NEIGHBORHOOD, FOR EXAMPLE, 'EL GUINARDÓ'.

Filter by Neighborhood: Horta-Guinardó

    SELECT * FROM parks_and_gardens WHERE neighborhood_name ILIKE '%el Guinardó%' LIMIT 10;

THIS QUERY FILTERS PARKS AND GARDENS BY NAME, FOR EXAMPLE, 'JARDINS DE CAN SENILLOSA'.

Filter by Name: Jardins de Can Senillosa

SELECT * FROM parks_and_gardens WHERE name ILIKE '%Jardins de Can Senillosa%' LIMIT 10;

THIS QUERY FILTERS PARKS AND GARDENS BY CREATED DATE, FOR EXAMPLE, '2020-01-10'.

Filter by Created Date: 2020-01-10

SELECT * FROM parks_and_gardens WHERE created = '2020-01-10' LIMIT 10;

THIS QUERY FILTERS PARKS AND GARDENS BY BOTH DISTRICT AND NAME, FOR EXAMPLE, DISTRICT 7 AND NAME 'ESPAI LÚDIC-ESPORTIU DE LA CAMPA'.

Filter by District 7 and Name: Espai Lúdic-Esportiu de la Campa

SELECT * FROM parks_and_gardens WHERE district_id = 7 AND name ILIKE '%Espai Lúdic-Esportiu de la Campa%' LIMIT 10;

THIS QUERY FILTERS PARKS AND GARDENS BY ZIP CODE, FOR EXAMPLE, '08035'.
